\begin{tcolorbox}[hbox, title=Reglas de derivación]
  \begin{minipage}{0.5\textwidth}
    \begin{description}
      \item[Suma]
      \[
        (f(x)+g(x))' = f'(x) + g'(x)
      \]
      \item[Diferencia]
      \[
        (f-g)'(x) = f'(x) - g'(x)
      \]
      \item[Producto]
      \[
        (fg)'(x) = f'(x)g(x) + f(x)g'(x)
      \]
      \item[Cociente]
      \[
        \left(\frac{f(x)}{g(x)}\right)' = \frac{f'(x)g(x) - f(x)g'(x)}{g(x)^2}
      \]
      \item[Regla de la cadena]
      \[
        (f(g))'(x) = f'(g(x))g'(x)
      \]
      \item[Inversa]
      \[
        (f^{-1})^\prime(x) = \frac{1}{f'(x)}
      \]
    \end{description}
  \end{minipage}
\end{tcolorbox}
